// Deploy as: Web App | Execute as: Me | Who has access: Anyone function doGet(e) { var action = (e.parameter && e.parameter.action) || 'ping'; if (action === 'ping') return ok('pong'); if (action === 'read') { var raw = readRaw(); if (!raw) { raw = { agents: [{id:1,name:'Super Admin',role:'superadmin',phone:'',email:'admin@company.com',pin:'Admin@1',password:'Admin@1',status:'Active',mustChangePIN:true,dailyTarget:0,managerId:null}], leads:[], calls:[], meetings:[], deletedIds:[], salesTeam:[], serviceTeam:[], nA:2, nL:1, nC:1, nM:1 }; } return ok(JSON.stringify(raw)); } return ok('unknown'); } function doPost(e) { try { var data = JSON.parse(e.postData.contents); var action = data.action || ''; if (action === 'login') { return handleLogin(data); } if (action === 'add_user' || action === 'edit_user') { return handleSaveUser(data); } if (action === 'change_password') { return handleChangePassword(data); } if (action === 'read') { var raw = readRaw(); return ok(JSON.stringify(raw)); } // Default: save full db writeAll(data); return ok('saved'); } catch(err) { return ok('error: ' + err.message); } } function handleLogin(data) { var email = (data.email || '').toLowerCase().trim(); var password = (data.password || '').trim(); var raw = readRaw(); if (!raw || !raw.agents) { // No data yet — allow Super Admin with default credentials if ((email === 'admin@dovecrm.local' || email === 'admin') && password === 'Admin@1') { var token = generateToken(); return ok(JSON.stringify({ token: token, user: { id:1, name:'Super Admin', role:'superadmin', email:'admin@dovecrm.local', phone:'', managerId:null, dailyTarget:0 } })); } return ok('error:invalid_credentials:3'); } var agents = raw.agents || []; var agent = null; for (var i = 0; i < agents.length; i++) { var a = agents[i]; if (a.email && a.email.toLowerCase() === email) { agent = a; break; } } if (!agent) return ok('error:invalid_credentials:3'); if (agent.status !== 'Active') return ok('error:account_inactive'); var pinMatch = agent.pin && agent.pin === password; var pwMatch = agent.password && agent.password === password; if (!pinMatch && !pwMatch) return ok('error:invalid_credentials:3'); var token = generateToken(); return ok(JSON.stringify({ token: token, user: { id: agent.id, name: agent.name, role: agent.role, email: agent.email || '', phone: agent.phone || '', managerId: agent.managerId || null, dailyTarget: agent.dailyTarget || 0 } })); } function handleSaveUser(data) { var raw = readRaw() || { agents:[], leads:[], calls:[], meetings:[], deletedIds:[] }; var user = data.user || {}; var agents = raw.agents || []; var existing = -1; for (var i = 0; i < agents.length; i++) { if (agents[i].id == user.id) { existing = i; break; } } if (existing >= 0) { // Edit — only update password if provided var old = agents[existing]; agents[existing] = Object.assign({}, old, user); if (!user.password) agents[existing].password = old.password; } else { // New user if (!user.password || user.password.length < 8) return ok('error:password_too_short'); agents.push(user); } raw.agents = agents; writeAll(raw); return ok('saved'); } function handleChangePassword(data) { var raw = readRaw(); if (!raw || !raw.agents) return ok('error:no_data'); var agents = raw.agents; for (var i = 0; i < agents.length; i++) { if (agents[i].id == data.userId) { var cur = agents[i].password || agents[i].pin || ''; if (cur !== data.currentPassword) return ok('error:wrong_current_password'); if (!data.newPassword || data.newPassword.length < 8) return ok('error:password_too_short'); agents[i].password = data.newPassword; agents[i].pin = data.newPassword; raw.agents = agents; writeAll(raw); return ok('password_changed'); } } return ok('error:user_not_found'); } function generateToken() { return 'tok_' + Math.random().toString(36).substr(2) + Date.now().toString(36); } function ok(msg) { return ContentService.createTextOutput(msg).setMimeType(ContentService.MimeType.JSON); } function doOptions(e) { return ContentService.createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } function getSheet(name) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sh = ss.getSheetByName(name); if (!sh) sh = ss.insertSheet(name); return sh; } function readRaw() { var sh = getSheet('_Data'); var rows = sh.getDataRange().getValues(); if (!rows || rows.length < 2) return null; var dataIdx = rows[0].indexOf('data'); if (dataIdx < 0) return null; try { return JSON.parse(rows[rows.length - 1][dataIdx]); } catch(e) { return null; } } function writeAll(data) { var ds = getSheet('_Data'); var ts = new Date().toISOString(); if (ds.getLastRow() === 0) ds.appendRow(['timestamp', 'data']); ds.appendRow([ts, JSON.stringify(data)]); if (ds.getLastRow() > 52) ds.deleteRows(2, ds.getLastRow() - 52); writeLeads(data.leads || [], data.agents || []); writeCalls(data.calls || [], data.agents || []); writeMeetings(data.meetings || [], data.agents || []); writeUsers(data.agents || []); } function agentName(agents, id) { if (!id) return 'Unassigned'; for (var i = 0; i < agents.length; i++) { if (agents[i].id == id) return agents[i].name; } return 'Unknown'; } function writeLeads(leads, agents) { var sh = getSheet('Leads'); sh.clearContents(); sh.appendRow(['ID','Name','Company','Phone','Email','Priority','Status','Assigned To','Source','Tags','Attempts','Last Called']); for (var i = 0; i < leads.length; i++) { var l = leads[i]; sh.appendRow([l.id, l.name, l.company||'', l.phone||'', l.email||'', l.priority||'', l.status||'', agentName(agents, l.assignedTo), l.source||'', (l.tags||[]).join(', '), l.attempts||0, l.lastCalled||'']); } formatHeader(sh); } function writeCalls(calls, agents) { var sh = getSheet('Calls'); sh.clearContents(); sh.appendRow(['ID','Lead Name','Telecaller','Date & Time','Call Type','Outcome','Duration (min)','Meeting Fixed','Follow-up Date','Notes']); for (var i = 0; i < calls.length; i++) { var c = calls[i]; sh.appendRow([c.id, c.leadName||'', c.agentName||agentName(agents,c.agentId), c.datetime||'', c.callType||'Customer Call', c.outcome||'', c.duration||0, c.meetingFixed?'Yes':'No', c.followupDate||'', c.notes||'']); } formatHeader(sh); } function writeMeetings(meetings, agents) { var sh = getSheet('Meetings'); sh.clearContents(); sh.appendRow(['ID','Lead Name','Telecaller','Date','Time','Mode','Location','Status','Outcome','Deal Value (Rs)','Follow-up Date','Notes']); for (var i = 0; i < meetings.length; i++) { var m = meetings[i]; sh.appendRow([m.id, m.leadName||'', m.agentName||agentName(agents,m.agentId), m.date||'', m.time||'', m.mode||'', m.location||'', m.status||'', m.meetingOutcome||'', m.dealValue||'', m.followupDate||'', m.notes||'']); } formatHeader(sh); } function writeUsers(agents) { var sh = getSheet('Users'); sh.clearContents(); sh.appendRow(['ID','Name','Role','Phone','Email','Status','Daily Target']); for (var i = 0; i < agents.length; i++) { var a = agents[i]; sh.appendRow([a.id, a.name, a.role||'', a.phone||'', a.email||'', a.status||'Active', a.dailyTarget||60]); } formatHeader(sh); } function formatHeader(sh) { var hRange = sh.getRange(1, 1, 1, sh.getLastColumn()); hRange.setBackground('#1e3a5f'); hRange.setFontColor('#ffffff'); hRange.setFontWeight('bold'); sh.setFrozenRows(1); sh.autoResizeColumns(1, sh.getLastColumn()); }
